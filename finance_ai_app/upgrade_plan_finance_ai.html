Global exception handler thêm
Hiện tại unhandled exception = app crash im lặng. Thêm sys.excepthook để log ra file và hiện dialog thân thiện.
File mới: app/core/error_handler.py
# error_handler.py — file mới: import sys, traceback, logging from pathlib import Path from PyQt6.QtWidgets import QMessageBox LOG_FILE = Path("data/error.log") def setup_global_handler(app): def handle_exception(exc_type, exc_value, exc_tb): if issubclass(exc_type, KeyboardInterrupt): sys.__excepthook__(exc_type, exc_value, exc_tb) return tb_str = "".join(traceback.format_exception(exc_type, exc_value, exc_tb)) logging.error(f"Unhandled exception:\n{tb_str}") with LOG_FILE.open("a") as f: f.write(tb_str) QMessageBox.critical(None, "Lỗi không mong muốn", f"Đã xảy ra lỗi. Chi tiết đã lưu vào:\n{LOG_FILE}\n\n{exc_value}") sys.excepthook = handle_exception # main.py — gọi ngay sau tạo QApplication: from app.core.error_handler import setup_global_handler setup_global_handler(app)

# nlp_parser.py — regex nhanh, không cần AI: import re from datetime import date AMOUNT_PATTERN = r'(\d+(?:[.,]\d+)?)\s*(k|nghìn|triệu|tr|đ|đồng)?' def parse_quick_add(text: str) -> dict | None: """ Input: "ăn phở 45k" Output: {amount:45000, desc:"ăn phở", type:"expense", date:today} """ m = re.search(AMOUNT_PATTERN, text, re.IGNORECASE) if not m: return None raw, unit = m.group(1), (m.group(2) or "").lower() amount = float(raw.replace(",", ".")) if unit in ("k", "nghìn"): amount *= 1000 elif unit in ("triệu", "tr"): amount *= 1000000 desc = re.sub(AMOUNT_PATTERN, "", text).strip() return {"amount": amount, "description": desc, "type": "expense", "date": str(date.today())} # chatbot_frame.py — detect và confirm với user: parsed = parse_quick_add(user_text) if parsed: self._add_assistant_message( f"Tôi hiểu: chi {parsed['amount']:,.0f}đ cho '{parsed['description']}'. Thêm vào không?" )
